let jsonObj = [
  { "Monday": "partly cloudy", "Tuesday": "sunny", "Wednesday": "rainy" },
  { "meat": "fish", "poultry": "eggs", "cleaners": "dishwashing liquid" }
]

let weekDays = jsonObj[0]
let groceries = jsonObj[1]
firstDay = Object.entries(weekDays)[0][0]
firstGroc = Object.entries(groceries)[0][0]

document.getElementById("demo").innerHTML = "On " + firstDay +
  " the weather was " + jsonObj[0].Monday + " and for " + firstGroc +
  " we had " + jsonObj[1].meat + "."
